## Config files

### ESLint

```js
// .eslintrc.js

module.exports = {
  extends: '@it-incubator/eslint-config',
  overrides: [
    {
      files: ['**/*.stories.tsx'],
      rules: {
        'react-hooks/rules-of-hooks': 'off',
        'no-console': 'off',
      },
    },
  ],
}
```

### Prettier

```js
// .prettierrc.js

module.exports = {
  ...require('@it-incubator/prettier-config'),
  //override settings here
}
```

### Stylelint

```js
// .stylelintrc.js

module.exports = {
  extends: '@it-incubator/stylelint-config',
}
```

### Prettier in package.json

```json
// package.json

{
  "prettier": "@it-incubator/prettier-config"
}
```
